# 从类型中创建类型

### 泛型类型
+ 泛型约束
  ```TypeScript
  interface Constraint {
    length: number
  }
  function genericConstraint<T>(arg: T extends Constraint): T {
    console.log(arg.length)
    return arg
  }
  genericConstraint('hello')
  ```
+ 在泛型约束中使用类型参数
  ``` TypeScript
  function getProperty<O, K extends keyof O>(obj: O, key: K) {
    return obj[key]
  }
  const obj = {
    a: 1,
    b: 2,
    c: 3
  }
  getProperty(obj, 'a')
  getProperty(obj, 'm') // 报错
  ```

### keyof 类型操作符
```TypeScript
  type Point = { x:number; y: number; z: number };
  type P = keyof Point; // x|y
  const p1: P = 'x';
  const p2: P = 'y';

  ---
  type Arrayish = { [i: number]: any }
  type A = keyof Arrayish // number
  const a1: A = 0

  ---
  type Mapish = { [key: string]: boolean }
  type M = keyof Mapish // string | number
  const m1: M = '1'
  const m2: M = 2
  const m3: M = true // 报错
```
